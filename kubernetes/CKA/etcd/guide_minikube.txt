backup & restore etcd:
Create a Snapshot

ETCDCTL_API=3 etcdctl \
  --endpoints=https://127.0.0.1:2379 \
  --cacert=/path/to/ca.crt \
  --cert=/path/to/server.crt \
  --key=/path/to/server.key \
  snapshot save snapshot.db

Restore from Snapshot

ETCDCTL_API=3 etcdctl \
  --endpoints=https://127.0.0.1:2379 \
  --cacert=/path/to/ca.crt \
  --cert=/path/to/server.crt \
  --key=/path/to/server.key \
  snapshot restart snapshot.db
